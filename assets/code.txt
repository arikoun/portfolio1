    /* SHOW MEDIA */
    function showMedia(projectName)
    {

        const project = projects[projectName];

        project.container.innerHTML = "";

        const file = project.media[project.currentMedia];


        /* VIDEO */
        if (file.endsWith(".mp4"))
        {

            const video = document.createElement("video");

            video.src = file;

            video.autoplay = true;
            video.muted = true;
            video.loop = true;
            video.controls = true;

            video.classList.add("media");

            project.container.appendChild(video);
        }


        /* IMAGE */
        else
        {

            const img = document.createElement("img");

            img.src = file;

            img.classList.add("media");

            project.container.appendChild(img);
        }
    }


    /* NEXT */
    function nextMedia(projectName)
    {

        const project = projects[projectName];

        project.currentMedia++;

        if (project.currentMedia >= project.media.length)
        {
            project.currentMedia = 0;
        }

        showMedia(projectName);
    }


    /* PREVIOUS */
    function previousMedia(projectName)
    {

        const project = projects[projectName];

        project.currentMedia--;

        if (project.currentMedia < 0)
        {
            project.currentMedia = project.media.length - 1;
        }

        showMedia(projectName);
    }

    /* LOAD FIRST MEDIA */
    showMedia("projectOne");
    showMedia("projectTwo");
    showMedia("projectThree");
    showMedia("projectfour");
    showMedia("projectfive");
